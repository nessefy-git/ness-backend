version: "3"

services:
  nginx-proxy:
    image: nginxproxy/nginx-proxy:latest
    container_name: nginx-proxy
    restart: always
    ports:
      - "80:80"
      - "443:443"
    volumes:
      - ./nginx/certs:/etc/nginx/certs:ro
      - ./nginx/vhost.d:/etc/nginx/vhost.d
      - ./nginx/html:/usr/share/nginx/html
      - /var/run/docker.sock:/tmp/docker.sock:ro
    networks:
      - proxy

  nginx-proxy-acme:
    image: nginxproxy/acme-companion
    container_name: nginx-proxy-acme
    restart: always
    environment:
      - DEFAULT_EMAIL=nessefy.officialmail@gmail.com
    volumes_from:
      - nginx-proxy
    volumes:
      - ./nginx/certs:/etc/nginx/certs
      - ./nginx/vhost.d:/etc/nginx/vhost.d
      - ./nginx/html:/usr/share/nginx/html
      - ./nginx/acme:/etc/acme.sh
      - /var/run/docker.sock:/var/run/docker.sock:ro
    networks:
      - proxy

  backend:
    image: nessefydockerhub/nessefy-backend:latest
    container_name: backend
    restart: always
    environment:
      - VIRTUAL_HOST=backend.nessefy.com
      - LETSENCRYPT_HOST=backend.nessefy.com
      - LETSENCRYPT_EMAIL=nessefy.officialmail@gmail.com
    expose:
      - "5000"
    networks:
      - proxy

networks:
  proxy:
    external: false
